# web_dnd

# клонировать проект:
git clone ....

# установить необходимые библиотеки:
pip install -r requirements.txt

# или каждую отдельно:
pip install django
.....

# перенести данные из DB:
python3 manage.py migrate

# создать суперюзера
python3 manage.py createsuperuser

# запустить сервер:
python3 manage.py runserver

# открыть:
http://127.0.0.1:8000

# админка:
http://127.0.0.1:8000/admin
